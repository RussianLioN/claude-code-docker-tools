#### –†–µ–∞–ª–∏–∑–∞—Ü–∏—è Sync Functions

```bash
# –î–æ–±–∞–≤–∏—Ç—å –≤ ai-assistant.zsh –ø–æ—Å–ª–µ —Å—Ç—Ä–æ–∫–∏ 175

sync_claude_config_in() {
  local lock_file="/tmp/claude-config-sync.lock"
  local source="$HOME/.claude"
  local target="$CLAUDE_STATE_DIR"

  # CRITICAL: Also sync ~/.claude.json (–≤ –∫–æ—Ä–Ω–µ ~/)
  local source_root_config="$HOME/.claude.json"
  local target_root_config="$CLAUDE_STATE_DIR/.claude.json"

  mkdir -p "$target"

  # Sync with flock (race condition protection)
  (
    flock -x -w 10 200 || {
      echo "‚ö†Ô∏è Timeout waiting for sync lock" >&2
      return 0
    }

    # 1. Sync ~/.claude/ directory
    if [[ -d "$source" ]]; then
      rsync -a --delete \
        --exclude 'debug/' \
        --exclude 'shell-snapshots/' \
        --exclude 'statsig/' \
        --exclude '.DS_Store' \
        "$source/" "$target/" 2>/dev/null || {
          echo "‚ö†Ô∏è Sync-in failed, using existing state" >&2
        }
    fi

    # 2. CRITICAL: Sync ~/.claude.json from HOME
    if [[ -f "$source_root_config" ]]; then
      cp "$source_root_config" "$target_root_config" 2>/dev/null || {
        echo "‚ö†Ô∏è Failed to sync ~/.claude.json" >&2
      }
    else
      echo "‚ö†Ô∏è ~/.claude.json not found - authentication may fail!" >&2
    fi

  ) 200>"$lock_file"
}

sync_claude_config_out() {
  local lock_file="/tmp/claude-config-sync.lock"
  local source="$CLAUDE_STATE_DIR"
  local target="$HOME/.claude"

  local source_root_config="$CLAUDE_STATE_DIR/.claude.json"
  local target_root_config="$HOME/.claude.json"

  # Skip in sandbox mode
  if [[ -n "$TRAE_SANDBOX_MODE" ]]; then
    return 0
  fi

  mkdir -p "$target"

  (
    flock -x -w 10 200 || {
      echo "‚ö†Ô∏è Timeout, skipping sync-out" >&2
      return 0
    }

    # 1. Sync directory back
    if [[ -d "$source" ]]; then
      rsync -a --delete \
        --exclude 'debug/' \
        --exclude 'shell-snapshots/' \
        --exclude 'statsig/' \
        --exclude '.DS_Store' \
        "$source/" "$target/" 2>/dev/null || {
          echo "‚ö†Ô∏è Sync-out failed" >&2
        }
    fi

    # 2. CRITICAL: Sync ~/.claude.json back
    if [[ -f "$source_root_config" ]]; then
      cp "$source_root_config" "$target_root_config" 2>/dev/null || {
        echo "‚ö†Ô∏è Failed to sync ~/.claude.json back" >&2
      }
    fi

  ) 200>"$lock_file"
}
```
